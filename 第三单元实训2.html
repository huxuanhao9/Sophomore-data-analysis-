import numpy as np
import matplotlib.pyplot as plt
plt.rcParams['font.sans-serif'] = 'SimHei'
plt.rcParams['axes.unicode_minus'] = False
text = np.load("D:/Desktop/data.npy",allow_pickle = True)
test = np.load("D:/Desktop/feature_names.npy",allow_pickle = True)
data = text[:20,1:]
data2 = data[::-1]
rk = ([list(data2[:,i]) for i in range(0,5)])
name = test[1:]
p=plt.figure(figsize=(20,20))
ax1 = p.add_subplot(3,1,1)
plt.boxplot(rk,notch=True,labels=name,meanline=True)
plt.xlabel('时间')
plt.ylabel('人口(万人)')
plt.legend(['年末总人口(万人)','男性人口(万人)','女性人口(万人)','城镇人口(万人)','乡村人口(万人)' ])
plt.title('1996-2015年人口数据特征间的关系(总图)')
plt.savefig("D:/Desktop/1996-2015年人口数据特征间的关系(总图).png")
plt.show()
